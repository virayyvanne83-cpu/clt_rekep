def stage3_subgoal_constraint1(end_effector, keypoints):
    """Ensure the red pen is upright (parallel to the z-axis). Use keypoints 4 and 5 to define the pen's orientation."""
    pen_vector = keypoints[5] - keypoints[4]
    z_axis = np.array([0, 0, 1])
    cost = 1 - np.dot(pen_vector / np.linalg.norm(pen_vector), z_axis)
    return cost

def stage3_subgoal_constraint2(end_effector, keypoints):
    """Ensure the red pen is positioned 10cm above the black pen holder opening (keypoint 7)."""
    offsetted_point = keypoints[7] + np.array([0, 0, 0.1])  # 10cm above keypoint 7
    cost = np.linalg.norm(keypoints[4] - offsetted_point)
    return cost
