def stage3_subgoal_constraint1(end_effector, keypoints):
    """Ensure the red pen is upright (parallel to the z-axis)."""
    pen_vector = keypoints[4] - keypoints[2]  # Vector along the red pen
    z_axis = np.array([0, 0, 1])  # Reference z-axis
    cost = np.abs(np.dot(pen_vector / np.linalg.norm(pen_vector), z_axis) - 1)
    return cost

def stage3_subgoal_constraint2(end_effector, keypoints):
    """Ensure the red pen is positioned 10cm above the black pen holder opening (keypoint 6)."""
    offsetted_point = keypoints[6] + np.array([0, 0, 0.1])  # 10cm above keypoint 6
    cost = np.linalg.norm(keypoints[4] - offsetted_point)
    return cost
